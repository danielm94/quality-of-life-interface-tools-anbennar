#SCOPE = COUNTRY
calculate_building_action_cost = {
	set_variable = {
		which = base_cost
		value = $base_cost$
	}
	set_variable = {
		which = $var_name$
		which = base_cost
	}
	multiply_variable = {
		which = $var_name$
		which = $build_cost_var$
	}
	change_variable = {
		which = $var_name$
		which = base_cost
	}
}
